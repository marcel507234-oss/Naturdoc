${text}

`; return; } const lines = text.split('\n').filter(line => line.trim() !== ''); lines.forEach(line => { const match = line.match(/^(.*?):\s*(.*?)\.\s*(.*)$/); if (match) { resultsDiv.innerHTML += `

${match[1].trim()}

${match[2].trim()}.

${match[3].trim()}

`; } else { resultsDiv.innerHTML += `

${line}

`; } }); } // Fehlerfall: Google hat geantwortet, aber mit einem Fehler else { let errorMessage = "Unbekannter Fehler von Google."; if (data.error && data.error.message) { errorMessage = data.error.message; // Hier holen wir den echten Grund } console.error("API Error Details:", data); resultsDiv.innerHTML = `

`; } } catch (error) { statusDiv.style.display = 'none'; resultsDiv.innerHTML = `
